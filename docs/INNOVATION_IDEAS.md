```typescript
// src/lib/analytics/recommendations.ts

interface Recommendation {
  id: string;
  type: 'inventory' | 'promotion' | 'customer' | 'bundling' | 'pricing';
  priority: 'urgent' | 'high' | 'medium' | 'low';
  title: string;
  description: string;
  impact: string; // "예상 매출 +500만원"
  actions: Action[];
  data: any; // 근거 데이터
}

interface Action {
  label: string;
  type: 'create_coupon' | 'send_email' | 'order_stock' | 'create_bundle';
  params: any;
  apiEndpoint: string;
}

export async function getRecommendations(): Promise<Recommendation[]> {
  const stats = await getDashboardStats();
  const recommendations: Recommendation[] = [];
  
  // 1. 재고 알림
  if (stats.lowStockProducts.length > 0) {
    stats.lowStockProducts.forEach(product => {
      recommendations.push({
        id: `stock-${product.id}`,
        type: 'inventory',
        priority: product.stock <= 5 ? 'urgent' : 'high',
        title: `"${product.name}" 품절 임박`,
        description: `현재 재고: ${product.stock}개. 평균 판매 속도를 고려하면 ${Math.floor(product.stock / 2)}일 후 품절 예상`,
        impact: '품절 시 일 500만원 매출 손실 예상',
        actions: [
          {
            label: '긴급 발주하기',
            type: 'order_stock',
            params: { productId: product.id, quantity: 50 },
            apiEndpoint: '/api/admin/inventory/order'
          },
          {
            label: '재입고 알림 발송',
            type: 'send_email',
            params: { type: 'restock', productId: product.id },
            apiEndpoint: '/api/admin/notifications/restock'
          }
        ],
        data: { product }
      });
    });
  }
  
  // 2. 시간대별 프로모션
  const peakHour = stats.hourlyOrders.reduce((max, curr) => 
    curr.count > max.count ? curr : max
  );
  
  if (peakHour.count > 10) {
    recommendations.push({
      id: 'peak-hour-promo',
      type: 'promotion',
      priority: 'high',
      title: `${peakHour.hour}시 주문 폭증 (${peakHour.count}건)`,
      description: `이 시간대에 타겟 프로모션을 진행하면 전환율 향상 예상`,
      impact: '예상 매출 +300만원/주',
      actions: [
        {
          label: `${peakHour.hour}시 특가 쿠폰 만들기`,
          type: 'create_coupon',
          params: {
            name: `${peakHour.hour}시 타임특가`,
            discountType: 'percentage',
            discountValue: 20,
            validHours: [peakHour.hour, peakHour.hour + 1]
          },
          apiEndpoint: '/api/admin/coupons/create'
        }
      ],
      data: { peakHour }
    });
  }
  
  // 3. 이탈 위험 VIP 고객
  const vipCustomers = await getVIPCustomers();
  const dormantVIPs = vipCustomers.filter(c => 
    daysSinceLastPurchase(c.lastPurchase) > 30
  );
  
  if (dormantVIPs.length > 0) {
    recommendations.push({
      id: 'dormant-vips',
      type: 'customer',
      priority: 'high',
      title: `VIP 고객 ${dormantVIPs.length}명 이탈 위험`,
      description: `30일 이상 미구매 VIP 고객에게 특별 혜택 제공 권장`,
      impact: `예상 복귀율 40% → 매출 +${dormantVIPs.length * 500000}원`,
      actions: [
        {
          label: 'VIP 전용 30% 쿠폰 발송',
          type: 'send_email',
          params: {
            customerIds: dormantVIPs.map(c => c.id),
            template: 'vip_comeback',
            couponValue: 30
          },
          apiEndpoint: '/api/admin/marketing/vip-campaign'
        }
      ],
      data: { dormantVIPs }
    });
  }
  
  // 4. 번들링 기회
  const bundleOpportunities = await findBundleOpportunities();
  
  if (bundleOpportunities.length > 0) {
    const topBundle = bundleOpportunities[0];
    recommendations.push({
      id: 'bundle-opportunity',
      type: 'bundling',
      priority: 'medium',
      title: `"${topBundle.product1}" + "${topBundle.product2}" 함께 구매 많음`,
      description: `${topBundle.frequency}번 함께 구매됨. 번들 상품 만들면 객단가 +35% 예상`,
      impact: '예상 매출 +200만원/월',
      actions: [
        {
          label: '번들 상품 자동 생성',
          type: 'create_bundle',
          params: {
            products: [topBundle.product1Id, topBundle.product2Id],
            discount: 15
          },
          apiEndpoint: '/api/admin/products/create-bundle'
        }
      ],
      data: { topBundle }
    });
  }
  
  return recommendations.sort((a, b) => {
    const priorityOrder = { urgent: 0, high: 1, medium: 2, low: 3 };
    return priorityOrder[a.priority] - priorityOrder[b.priority];
  });
}
```

```typescript
// src/lib/recommendations/customer.ts

interface CustomerProfile {
  userId: string;
  petInfo: {
    name: string;
    breed: string;
    size: 'small' | 'medium' | 'large';
    age: number;
  };
  purchaseHistory: Order[];
  browsingHistory: ProductView[];
  preferences: {
    favoriteCategories: string[];
    priceRange: [number, number];
    favoriteColors: string[];
  };
}

export async function getPersonalizedFeed(
  userId: string
): Promise<RecommendationSection[]> {
  const profile = await getCustomerProfile(userId);
  const sections: RecommendationSection[] = [];
  
  // 1. AI 추천 (구매 이력 기반)
  const aiRecommendations = await getAIRecommendations(profile);
  sections.push({
    title: `${profile.petInfo.name}님을 위한 추천`,
    products: aiRecommendations,
    reason: '지난 구매 이력 기반'
  });
  
  // 2. 품종별 인기 상품
  const breedPopular = await getBreedPopularProducts(profile.petInfo.breed);
  sections.push({
    title: `${profile.petInfo.breed}에게 인기`,
    products: breedPopular
  });
  
  // 3. 함께 구매한 상품
  if (profile.purchaseHistory.length > 0) {
    const lastPurchase = profile.purchaseHistory[0];
    const bundleProducts = await getFrequentlyBoughtTogether(lastPurchase);
    sections.push({
      title: '함께 구매하면 좋은 상품',
      products: bundleProducts
    });
  }
  
  // 4. 가격대 맞춤
  const priceMatchProducts = await getProductsInPriceRange(
    profile.preferences.priceRange
  );
  sections.push({
    title: '내 예산에 딱 맞는',
    products: priceMatchProducts
  });
  
  return sections;
}
```
